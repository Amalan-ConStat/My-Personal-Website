# load the data
emperors <- readr::read_csv("https://raw.githubusercontent.com/rfordatascience/tidytuesday/master/data/2019/2019-08-13/emperors.csv")

# load packages
library(ggplot2)
library(dplyr)
library(gganimate)

# load theme package
library(ggthemr)
ggthemr("flat")

Birth Province

emperors %>%
  group_by(birth_prv) %>%
  count(sort = TRUE) %>%
  ggplot(.,aes(birth_prv,n,label=n))+geom_col()+
  xlab("Birth Province")+ylab("Frequency")+geom_text(vjust=1)+
  theme(axis.text.x = element_text(angle=30))

ggplot(emperors ,aes(stringr::str_wrap(birth_prv,12),stringr::str_wrap(rise,15)))+geom_jitter()+
  xlab("Birth Province")+ylab("Risen to Power")+
  theme(axis.text.x = element_text(angle=30))

emperors %>%
  mutate(reign_time=reign_end-reign_start) %>%
ggplot(.,aes(stringr::str_wrap(birth_prv,12),abs(reign_time)))+
  geom_jitter()+
  xlab("Birth Province")+ylab("Days in Power")+
  theme(axis.text.x = element_text(angle=30))
